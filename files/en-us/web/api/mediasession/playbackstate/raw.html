The playbackState property of the {{domxref("MediaSession")}} interface indicates whether the current media session is playing.

Syntax

var playbackState = MediaSession.playbackState
MediaSession.playbackState = playbackState

Value

One of "none", "paused", or "playing".

Example

The following example sets up event handlers, for pausing and playing:

var audio = document.querySelector("#player");
audio.src = "song.mp3";

navigator.mediaSession.setActionHandler('play', play);
navigator.mediaSession.setActionHandler('pause', pause);

function play() {
   audio.play();
   navigator.mediaSession.playbackState = "playing";
}

function pause() {
   audio.pause();
   navigator.mediaSession.playbackState = "paused";
}
