Retorna el total de espacio vertical disponible en la pantalla.

Sintaxis

iAvail = window.screen.availHeight

Ejemplo

if (window.screen.availHeight !== window.screen.height) {
  // algo está ocurriendo!
  // use la altura disponible para dimensionar la ventana
}
