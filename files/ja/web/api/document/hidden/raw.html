Document.hidden は読み込み専用のプロパティで、ページが隠されていると見なされるかどうかを示すブーリアンの値を返します。

構文

var boolean = document.hidden

例

document.addEventListener("visibilitychange", function() {
  console.log( document.hidden );
  // 振る舞いを修正...
});
