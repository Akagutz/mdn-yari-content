@@iterator プロパティの初期値は、{{jsxref("Array.prototype.values")}} プロパティの初期値と同じ function オブジェクトです。

構文

arguments[Symbol.iterator]()

例

for...of ループを使用した繰り返し

function f() {
  // ブラウザが for..of ループと
  // ループ内での let スコープ変数をサポートしている必要があります。
  for (let letter of arguments) {
    console.log(letter);
  }
}
f('w', 'y', 'k', 'o', 'p');
