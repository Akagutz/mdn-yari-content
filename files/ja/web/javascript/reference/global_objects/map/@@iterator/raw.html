概要

@@iterator プロパティの初期値は{{jsxref("Map.prototype.entries()", "entries")}}プロパティの初期値と同じ関数オブジェクトです。

構文

myMap[Symbol.iterator]

例

例: [@@iterator]()を使う

var myMap = new Map();
myMap.set("0", "foo");
myMap.set(1, "bar");
myMap.set({}, "baz");

var mapIter = myMap[Symbol.iterator]();

console.log(mapIter.next().value); // ["0", "foo"]
console.log(mapIter.next().value); // [1, "bar"]
console.log(mapIter.next().value); // [Object, "baz"]

例: for..ofとともに[@@iterator]()を使う

var myMap = new Map();
myMap.set("0", "foo");
myMap.set(1, "bar");
myMap.set({}, "baz");

for (var v of myMap) {
  console.log(v);
}
